    # File lib/alimento/alimento_.rb, line 43
def aibc(indice)
    vector = []
    datos[indice][1..datos[indice].length-1].zip(datos[indice][0..datos[indice].length-2]) do | gi, gi_1 | 
        if gi < datos[indice][0]
            vector << 0.0
        else 
            vector << (((gi-datos[indice][0])+(gi_1-datos[indice][0]))/2)*5
        end
    end 
    
    vector.reduce(:+)
end
