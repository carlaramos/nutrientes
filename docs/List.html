def insert vector
    vector.each do |element|
        node = Node.new(element,nil,nil)
        if(@head==nil)
            @head = node
            @tail = node
        else
            node.prev=@tail
            @tail.next = node
            @tail=node
        end
    end
end

    Se ordena una lista utilizando el método each

    # File lib/alimento/list.rb, line 84
def ordenar_each
    vector = self.map { |x| x }
    indice=0
    vector.each do |x|
        aux = x
        c=indice
        indice2=indice+1
        
        vector[indice2..vector.length-1].each do |y|
            if aux > y 
                aux = y
                c = indice2
            end
            indice2+=1
        end
        
        vector[c] = x
        vector[indice] = aux
        
        indice+=1
    end
    vector
end

def ordenar_seleccion 
    vector = self.map { |x| x }
    for i in 0..self.count-1
        aux = vector[i]
        c = i

        for j in i + 1..self.count-1
            if aux > vector[j] 
                aux = vector[j]
                c = j 
            end
        end
        
        vector[c] = vector[i]
        vector[i] = aux
    end
    vector
end
